# === AWS ===
# AWS Access Key ID
AKIA[0-9A-Z]{16}
# AWS Secret Access Key (generic assignment)
(?i)(aws_secret_access_key|aws_secret_key)\s*[=:]\s*[A-Za-z0-9/+=]{40}

# === GitHub ===
# GitHub personal access tokens, OAuth, app tokens
gh[pousr]_[A-Za-z0-9_]{36,255}
# GitHub fine-grained tokens
github_pat_[A-Za-z0-9_]{22,255}

# === Google ===
# Google API key
AIza[0-9A-Za-z\-_]{35}

# === Stripe ===
# Stripe live secret/publishable keys
[sr]k_live_[A-Za-z0-9]{20,}

# === Private Keys ===
-----BEGIN\s?(RSA|EC|DSA|OPENSSH|PGP)?\s?PRIVATE KEY-----

# === Generic Tokens & Secrets ===
# Bearer tokens in code
(?i)bearer\s+[A-Za-z0-9\-._~+/]+=*
# Generic API key assignments
(?i)(api[_-]?key|apikey)\s*[=:]\s*["'][A-Za-z0-9]{16,}["']
# Generic secret assignments
(?i)(secret|secret[_-]?key)\s*[=:]\s*["'][A-Za-z0-9]{16,}["']
# Generic password assignments
(?i)(password|passwd|pwd)\s*[=:]\s*["'][^"']{8,}["']
# Generic token assignments
(?i)(token|auth[_-]?token|access[_-]?token)\s*[=:]\s*["'][A-Za-z0-9\-._]{16,}["']

# === Database Connection Strings ===
# MongoDB with credentials
mongodb(\+srv)?://[^/\s:]+:[^@/\s]+@
# PostgreSQL/MySQL with credentials
(?i)(postgres|mysql|mariadb)://[^/\s:]+:[^@/\s]+@

# === Slack ===
xox[bporas]-[A-Za-z0-9\-]{10,}

# === Twilio ===
SK[0-9a-fA-F]{32}

# === SendGrid ===
SG\.[A-Za-z0-9\-_]{22,}\.[A-Za-z0-9\-_]{22,}

# === Mailgun ===
key-[0-9a-zA-Z]{32}

# === npm ===
npm_[A-Za-z0-9]{36}

# === PyPI ===
pypi-[A-Za-z0-9]{16,}

# === JWT tokens (hardcoded) ===
eyJ[A-Za-z0-9-_]+\.eyJ[A-Za-z0-9-_]+\.[A-Za-z0-9-_]+

# === Azure ===
(?i)(azure[_-]?storage[_-]?key|azure[_-]?account[_-]?key)\s*[=:]\s*["'][A-Za-z0-9+/=]{44,}["']

# === Heroku ===
(?i)heroku[_-]?api[_-]?key\s*[=:]\s*["'][0-9a-fA-F\-]{36}["']
